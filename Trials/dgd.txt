func DFS(adjacencyList map[string][]string, currentVertex string, visited []string) []string {
	visited = append(visited, currentVertex)
	children := adjacencyList[currentVertex]
	if len(children) == 0 {
		return visited
	}

	for _, child := range children {
		if !contains(visited, child) {
			visited = DFS(adjacencyList, child, visited)
		}
	}

	return visited
}

func contains(arr []string, target string) bool {
	for _, elem := range arr {
		if elem == target {
			return true
		}
	}
	return false
}

func getName(info string) string {
	parts := strings.Split(info, " ")
	if len(parts) == 0 {
		return ""
	}
	return parts[0]
}



func Queue(paths [][]string, antsAmount int) []int {
	if len(paths) == 0 || antsAmount <= 0 {
		return nil
	}

	antsQueue := make([]int, len(paths))
	rooms := make([]int, len(paths))

	for i, path := range paths {
		rooms[i] = len(path)
	}

	for antsAmount > 0 {
		index := findLowestRoomCountIndex(rooms, antsQueue)
		antsQueue[index]++
		antsAmount--
	}

	return antsQueue
}

func findLowestRoomCountIndex(rooms []int, antsQueue []int) int {
	lowestValue := 10000
	lowestIndex := 0

	for i := range rooms {
		sum := rooms[i] + antsQueue[i]
		if sum < lowestValue {
			lowestValue = sum
			lowestIndex = i
		}
	}

	return lowestIndex
}
